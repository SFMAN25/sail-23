    <script>
        const canvas = document.getElementById('fireworks');
        const ctx = canvas.getContext('2d');
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;

        let startFireworks = false;

        class Firework {
            constructor(x, y, color) {
                this.x = x;
                this.y = y;
                this.color = color;
                this.particles = [];
                for (let i = 0; i < 70; i++) {
                    this.particles.push({
                        x: x,
                        y: y,
                        vx: (Math.random() - 0.5) * 9,
                        vy: (Math.random() - 0.5) * 9,
                        alpha: 1
                    });
                }
            }
            update() {
                this.particles.forEach(p => {
                    p.x += p.vx;
                    p.y += p.vy;
                    p.alpha -= 0.012;
                });
            }
            draw() {
                this.particles.forEach(p => {
                    ctx.fillStyle = this.color;
                    ctx.globalAlpha = p.alpha;
                    ctx.beginPath();
                    ctx.arc(p.x, p.y, 2.5, 0, Math.PI * 2);
                    ctx.fill();
                });
            }
        }

        let fireworks = [];
        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            if (startFireworks && Math.random() < 0.3) {
                fireworks.push(new Firework(
                    Math.random() * canvas.width, 
                    Math.random() * canvas.height * 0.65, 
                    `hsl(${Math.random() * 360}, 100%, 65%)`
                ));
            }
            
            fireworks.forEach((fw, i) => {
                fw.update();
                fw.draw();
                fw.particles = fw.particles.filter(p => p.alpha > 0);
                if (fw.particles.length === 0) fireworks.splice(i, 1);
            });
            requestAnimationFrame(animate);
        }
        animate();

        // دالة البدء المزامنة بالمللي ثانية مع صوت الأغنية الرسمي
        function startCelebration() {
            document.getElementById('start-overlay').style.display = 'none';
            
            const introCountdown = document.getElementById('intro-countdown');
            const birthdayContent = document.getElementById('birthday-content');
            const audio = document.getElementById('bg-music');
            
            introCountdown.style.display = 'block';
            
            // تشغيل الأغنية فوراً
            audio.play().catch(err => console.log("Audio playback error:", err));

            // المزامنة الذكية: نراقب وقت الأغنية الحالي ثانية بثانية لتحديث الرقم
            let lastTime = -1;
            let countdownEnded = false;

            const syncTimer = setInterval(() => {
                if (countdownEnded) return;

                // قراءة الوقت الحالي للأغنية بالثواني
                let currentTime = Math.floor(audio.currentTime);

                if (currentTime !== lastTime) {
                    lastTime = currentTime;
                    let timeLeft = 10 - currentTime; // لأن العد تنازلي من 10 ثواني

                    if (timeLeft >= 0) {
                        introCountdown.innerText = timeLeft;
                        // تأثير تكبير خفيف مع كل رقم جديد
                        introCountdown.style.transform = 'scale(1.2)';
                        setTimeout(() => {
                            introCountdown.style.transform = 'scale(1)';
                        }, 200);
                    } else {
                        // انتهى الـ 10 ثواني بالظبط مع صوت الأغنية
                        countdownEnded = true;
                        clearInterval(syncTimer);
                        
                        introCountdown.style.opacity = '0';
                        introCountdown.style.transform = 'scale(0)';
                        
                        setTimeout(() => {
                            introCountdown.remove();
                            // إظهار الكلام والترتيب الاحترافي في النص
                            birthdayContent.classList.add('show');
                            // إطلاق الألعاب النارية
                            startFireworks = true;
                        }, 300);
                    }
                }
            }, 50); // فحص سريع جداً كل 50 مللي ثانية لضمان دقة متناهية
        }

        window.addEventListener('resize', () => {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        });
    </script>
